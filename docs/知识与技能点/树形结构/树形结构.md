```javascript
creatTree(list: Array<any>) {
    const tree = [];
    const mapObj = {};
    list.forEach(item => {
      mapObj[item.code] = item;
      item.isLeaf = true;
      item.children = [];
    });
    list.forEach(item => {
      const parent = mapObj[item.parentCode];
      if (parent) {
        parent.isLeaf = false;
        parent.children.push(item);
      } else {
        item.isLeaf = item.children.length === 0;
        tree.push(item);
      }
    });
    console.log('处理后的树', tree);
    return tree;
 }
```
